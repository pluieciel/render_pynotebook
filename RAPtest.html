def step_midpoint(f,x0,y0,h):
    return y0+h*(f(x0+h/2,y0+h/2*f(x0,y0)))

def f_higher(b, k):
    def f(x, z):
        return -b*z[1]-k*z[0]
    return f

def g_higher(f):
    def g(x, z):
        return np.array([z[1], f(x, z)])
    return g

def integrate_mid(g, t_0, z0, dt, Nt):
    record = z0
    z = z0
    t = t_0
    for _ in range(Nt):
        t_new = t + dt
        z = step_midpoint(g, t, z, dt)
        t = t_new
        record = np.vstack((record, z))
    return record
